\section*{Chest}
\begin{itemize}[leftmargin=*]
  \item \textbf{Barbell Bench Press} (\textit{Intermediate})\\
        \textbf{Primary:} Chest $|$ \textbf{Secondary:} Anterior Deltoid, Triceps\\
        \textbf{Equipment:} Flat Olympic Press station, 7ft Olympic Barbell, Grip Weight Plates
  \item \textbf{Incline Barbell Press} (\textit{Intermediate})\\
        \textbf{Primary:} Upper Chest $|$ \textbf{Secondary:} Anterior Deltoid, Triceps\\
        \textbf{Equipment:} Incline Bench Press station, 7ft Olympic Barbell, Grip Weight Plates
  \item \textbf{Smith Machine Bench Press} (\textit{Beginner})\\
        \textbf{Primary:} Chest $|$ \textbf{Secondary:} Anterior Deltoid, Triceps\\
        \textbf{Equipment:} Smith Machine, Flat Bench, Grip Weight Plates
  \item \textbf{Cable Chest Fly (High-to-Low)} (\textit{Intermediate})\\
        \textbf{Primary:} Lower Chest $|$ \textbf{Secondary:} Anterior Deltoid\\
        \textbf{Equipment:} Dual Adjustable Pulley / Functional Trainer
  \item \textbf{Dumbbell Flat Press} (\textit{Intermediate})\\
        \textbf{Primary:} Chest $|$ \textbf{Secondary:} Anterior Deltoid, Triceps\\
        \textbf{Equipment:} Flat Bench, Dumbbells
  \item \textbf{Chest Fly Machine} (\textit{Beginner})\\
        \textbf{Primary:} Chest $|$ \textbf{Secondary:} Anterior Deltoid\\
        \textbf{Equipment:} Chest Fly / Rear Delt Machine
\end{itemize}
